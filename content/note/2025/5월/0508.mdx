zod 함수와 타입 네임스페이스가 같다(z.object() + z.infer<> - import { z } from 'zod';)

어떻게 구현?

```tsx
// types.ts
import { z } from "zod";
import { Timestone } from "../ValueObject";

export * from "zod";

//////// 이 아래로는 portlogics용 커스텀 스키마 및 타입 export해주기 ////////////
export const timestone = () =>
  z.custom<Timestone>((value: unknown) => value instanceof Timestone, {
    message: "유효한 Timestone 객체가 아닙니다",
  });
```

```tsx
// index.ts
import * as z from './types';
export { z };
```
